        (function () {
            function updateLinks() {
                var links = document.querySelector(".links ul");
                if (links) {
                    for (let i = 0; i < links.children.length - 1; i++) {
                        const current = links.children[i];
                        const next = links.children[i + 1];
                        if (current.offsetTop < next.offsetTop) {
                            current.classList.add('last-in-line');
                        } else {
                            current.classList.remove('last-in-line');
                        }
                    }
                }
            }
            updateLinks();
            window.addEventListener('resize', updateLinks);
            window.addEventListener('load', updateLinks);
        })();
